016    /**
017     * Assign to &lt;logout success-handler-ref=&quot;hander-reference&quot;/&gt; fot handle logout action.<br/>
018     * When logout from GWT RPC, just void message return.
019     * If logout from normal URL will redirect page to
020     * {@link #setLogoutSuccessUrl(java.lang.String) logoutSuccessUrl}.<br/>
021     * å»ºç??¬é??¥å¯¦ä¾‹ä¸¦è¨­å???lt;logout success-handler-ref=&quot;å¯¦ä??ƒç…§&quot;/&gt; ä»¥è??†ç™»?„ä?æ¥­ï?<br/>
022     * ?¶ä½¿?¨GWT RPC?²è??»å‡º?‚ï??¬é¡¯?¥æ??žæ?GWT RPCç¨‹å???br/>
023     * ?¥ä½¿?¨æ­£å¸¸å???™»?ºï??‡é??¢æ??žè??°{@link #setLogoutSuccessUrl(java.lang.String) logoutSuccessUrl}
024     */
025    public class GwtLogoutSuccessHandler implements LogoutSuccessHandler,
026            InitializingBean, ServletContextAware {
027    
028        private SimpleUrlLogoutSuccessHandler defaultHandler = new SimpleUrlLogoutSuccessHandler();
029        private String logoutSuccessUrl = "/";
030        private ServletContext servletContext;
031    
032        @Override
033        public void setServletContext(ServletContext servletContext) {
034            this.servletContext = servletContext;
035        }
036    
037        @Override
038        public void afterPropertiesSet() throws Exception {
039            Assert.notNull(logoutSuccessUrl, "logoutSuccessUrl must be specified");
040            defaultHandler.setDefaultTargetUrl(logoutSuccessUrl);
041        }
042    
043        public String getLogoutSuccessUrl() {
044            return logoutSuccessUrl;
045        }
046    
047        public void setLogoutSuccessUrl(String logoutSuccessUrl) {
048            this.logoutSuccessUrl = logoutSuccessUrl;
049            defaultHandler.setDefaultTargetUrl(logoutSuccessUrl);
050        }
051    
052        @Override
053        public void onLogoutSuccess(HttpServletRequest request, HttpServletResponse response, Authentication authentication) throws IOException, ServletException {
054            if (GwtResponseUtil.isGwt(request)) {
055                String responsePayload = String.format("//OK[[],%d,%s]", AbstractSerializationStream.DEFAULT_FLAGS,
056                        AbstractSerializationStream.SERIALIZATION_STREAM_VERSION);
057                GwtResponseUtil.writeResponse(servletContext, request, response, responsePayload);
058            } else {
059                defaultHandler.onLogoutSuccess(request, response, authentication);
060            }
061    
062        }
063    }

001    package com.gwt.ss;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import javax.servlet.http.HttpServletResponse;
005    import org.aspectj.lang.JoinPoint;
006    
007    public class HttpHolder {
008    
009        HttpServletRequest request;
010        HttpServletResponse response;
011    
012        public HttpServletRequest getRequest() {
013            return request;
014        }
015    
016        public void setRequest(HttpServletRequest request) {
017            this.request = request;
018        }
019    
020        public HttpServletResponse getResponse() {
021            return response;
022        }
023    
024        public void setResponse(HttpServletResponse response) {
025            this.response = response;
026        }
027    
028        public boolean isGwt() {
029            return GwtResponseUtil.isGwt(request);
030        }
031    
032        public static HttpHolder getInstance(JoinPoint jp) {
033            if (jp == null) {
034                return null;
035            } else {
036                HttpHolder holder = new HttpHolder();
037                for (Object obj : jp.getArgs()) {
038                    if (obj != null) {
039                        if (obj instanceof HttpServletRequest) {
040                            holder.setRequest((HttpServletRequest) obj);
041                        } else if (obj instanceof HttpServletResponse) {
042                            holder.setResponse((HttpServletResponse) obj);
043                        }
044                        if (holder.getRequest() != null && holder.getResponse() != null) {
045                            break;
046                        }
047                    }
048                }
049                return holder;
050            }
051        }
052    }

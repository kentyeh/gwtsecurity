095        private PayloadInfo extract(JoinPoint jp) throws IOException, ServletException {
096            PayloadInfo result = payloadHolder.get();
097            HttpHolder httpHolder = HttpHolder.getInstance(jp);
098            HttpServletRequest request = httpHolder.getRequest();
099            String username = null;
100            String password = null;
101            boolean rememberMe = false;
102            if (result == null && request != null) {
103                String payload = RPCServletUtils.readContentAsGwtRpc(request);
104                RPCRequest rpcRequest = RPC.decodeRequest(payload, null, serializationPolicyProvider);
105                Object[] requestParams = rpcRequest.getParameters();
106                assert requestParams.length > 1 : "parameter count incorrect";
107                username = (String) requestParams[0];
108                password = (String) requestParams[1];
109                if (requestParams.length > 2) {
110                    try {
111                        rememberMe = (Boolean) requestParams[2];
112                    } catch (Exception e) {
113                    }
114                }
115                if (username != null && password != null) {
116                    result = new PayloadInfo(username, password, httpHolder, rememberMe);
117                    payloadHolder.set(result);
118                }
119            }
120            return result;
121        }

123        @Around("execution(* org.springframework.security.web.authentication.AbstractAuthenticationProcessingFilter.doFilter(..))")
124        public Object doFilter(ProceedingJoinPoint pjp) throws Throwable {
125            HttpHolder httpHolder = HttpHolder.getInstance(pjp);
126            AbstractAuthenticationProcessingFilter filter = (AbstractAuthenticationProcessingFilter) pjp.getTarget();
127            if (httpHolder.isGwt() && requiresAuthentication(filter, httpHolder.getRequest())) {
128                PayloadInfo pi = null;
129                try {
130                    pi = extract(pjp);
131                    UsernamePasswordAuthenticationToken authRequest = new UsernamePasswordAuthenticationToken(pi.getUsername(), pi.getPassword());
132                    Class c = AbstractAuthenticationProcessingFilter.class;
133                    Field f = c.getDeclaredField("allowSessionCreation");
134                    f.setAccessible(true);
135                    boolean allowSessionCreation = (Boolean) f.get(filter);
136                    HttpSession session = httpHolder.getRequest().getSession(allowSessionCreation);
137                    if (session != null) {
138                        httpHolder.getRequest().getSession().setAttribute(GwtResponseUtil.SPRING_SECURITY_LAST_USERNAME_KEY,
139                                TextEscapeUtils.escapeEntities(pi.getUsername()));
140                    }
141                    authRequest.setDetails(filter.getAuthenticationDetailsSource().buildDetails(httpHolder.getRequest()));
142                    Authentication authResult = getAuthenticationManager().authenticate(authRequest);
143                    SecurityContextHolder.getContext().setAuthentication(authResult);
144                    filter.getRememberMeServices().loginSuccess(pi.isRemeberMe()
145                            ? new RemeberRequestWrapper(httpHolder.getRequest(), rememberMeParameter)
146                            : httpHolder.getRequest(), httpHolder.getResponse(), authResult);
147                    
148                    //Patch provided by Steven Jardine steven.j...@gmail.com http://code.google.com/u/@UhVVQ1JZAxVEXgF4/
149                    f = c.getDeclaredField("sessionStrategy");
150                    f.setAccessible(true);
151                    SessionAuthenticationStrategy sessionStrategy = (SessionAuthenticationStrategy) f.get(filter);
152                    sessionStrategy.onAuthentication(authResult, httpHolder.getRequest(), httpHolder.getResponse());
153                    
154                    applicationContext.publishEvent(new InteractiveAuthenticationSuccessEvent(authResult, this.getClass()));
155                    return null;
156                } catch (Exception e) {
157                    if (logger.isErrorEnabled()) {
158                        logger.error("Gwt login fail:", e);
159                    }
160                    GwtResponseUtil.processGwtException(servletContext, httpHolder.getRequest(), httpHolder.getResponse(), e);
161                    return null;
162                } finally {
163                    payloadHolder.remove();
164                }
165            } else {
166                return pjp.proceed();
167            }
168        }

170        protected boolean requiresAuthentication(AbstractAuthenticationProcessingFilter filter, HttpServletRequest request) {
171            String uri = request.getRequestURI();
172            if (logger.isDebugEnabled()) {
173                logger.debug("RequiresAuthentication check for \n"
174                        + "url = " + uri + "\ncontext path = " + request.getContextPath() + "\nprocessUrl = " + filter.getFilterProcessesUrl());
175            }
176            if (uri == null || uri.isEmpty()) {
177                return false;
178            } else {
179                int pathParamIndex = uri.indexOf(';');
180    
181                if (pathParamIndex > 0) {
182                    // strip everything after the first semi-colon
183                    uri = uri.substring(0, pathParamIndex);
184                }
185    
186                if ("".equals(request.getContextPath())) {
187                    return uri.endsWith(filter.getFilterProcessesUrl());
188                }
189                return uri.endsWith(request.getContextPath() + filter.getFilterProcessesUrl());
190            }
191        }

193        @Override
194        public void onApplicationEvent(InteractiveAuthenticationSuccessEvent event) {
195            PayloadInfo pi = payloadHolder.get();
196            if (pi != null && pi.getHttpHolder().isGwt()) {
197                HttpServletRequest request = pi.getHttpHolder().getRequest();
198                Authentication authResult = event.getAuthentication();
199                if (logger.isDebugEnabled()) {
200                    logger.debug("Gwt authentication success. Updating SecurityContextHolder to contain: " + authResult);
201                }
202                GwtResponseUtil.writeResponse(servletContext, request, pi.getHttpHolder().getResponse(),
203                        String.format("//OK[[],%s,%s]", AbstractSerializationStream.DEFAULT_FLAGS,
204                        AbstractSerializationStream.SERIALIZATION_STREAM_VERSION));
205            }
206        }

261        private class RemeberRequestWrapper extends HttpServletRequestWrapper {
262    
263            private String rememberMeParameter = "_spring_security_remember_me";
264    
265            public RemeberRequestWrapper(HttpServletRequest request, String rememberMeParameter) {
266                super(request);
267                if (rememberMeParameter != null && !rememberMeParameter.isEmpty()) {
268                    this.rememberMeParameter = rememberMeParameter;
269                }
270            }
271    
272            @Override
273            public String getParameter(String name) {
274                return name.equals(rememberMeParameter) ? "true" : super.getParameter(name);
275            }

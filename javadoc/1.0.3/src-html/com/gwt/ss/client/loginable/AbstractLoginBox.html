001    package com.gwt.ss.client.loginable;
002    
003    import com.google.gwt.event.shared.HandlerRegistration;
004    import com.google.gwt.user.client.ui.DialogBox;
005    
006    /**
007     *  Abstract dialogBox for {@link HasLoginHandler HasLoginHandler}. 
008     * @author Kent Yeh
009     */
010    public abstract class AbstractLoginBox extends DialogBox implements HasLoginHandler {
011    
012        public AbstractLoginBox(boolean autoHide, boolean modal) {
013            super(autoHide, modal);
014        }
015    
016        public AbstractLoginBox(boolean autoHide) {
017            super(autoHide);
018        }
019    
020        public AbstractLoginBox() {
021        }
022        
023        @Override
024        public HandlerRegistration addLoginHandler(LoginHandler handler) {
025            return super.addHandler(handler, LoginEvent.getType());
026        }
027    }

037        private void validate(TreeLogger logger, String typeName) throws UnableToCompleteException {
038            if (!typeName.endsWith("Async")) {
039                logger.log(TreeLogger.ERROR, format("Asynchronous service's name must ends with \"Async\",Obviously \"%s\" doesn't", typeName));
040                throw new UnableToCompleteException();
041            }
042            TypeOracle typeOracle = this.context.getTypeOracle();
043            this.typeName = typeName;
044            try {
045                this.sourceType = typeOracle.getType(typeName);
046                if (this.sourceType.isInterface() == null) {
047                    logger.log(TreeLogger.ERROR, format("%s is not an interface.", sourceType.getQualifiedSourceName()), null);
048                    throw new UnableToCompleteException();
049                }
050                JClassType loginableAsyncType = typeOracle.getType(LoginableAsync.class.getName());
051                if (!sourceType.getFlattenedSupertypeHierarchy().contains(loginableAsyncType)) {
052                    logger.log(TreeLogger.ERROR, format("Type: %s not extends LoginableAsync", typeName));
053                    throw new UnableToCompleteException();
054                }
055                this.packageName = this.sourceType.getPackage().getName();
056            } catch (NotFoundException ex) {
057                logger.log(TreeLogger.ERROR, format("validate %s fail", typeName), ex);
058                throw new UnableToCompleteException();
059            }
060            this.serviceName = typeName.substring(0, this.typeName.length() - 5);
061            this.serviceType = typeOracle.findType(serviceName);
062            if (serviceType == null) {
063                logger.log(TreeLogger.ERROR, format("Could not find remote service type : %s", serviceName));
064                throw new UnableToCompleteException();
065            }
066            if (this.serviceType.isInterface() == null) {
067                logger.log(TreeLogger.ERROR, format("%s is not an interface.", serviceType.getQualifiedSourceName()), null);
068                throw new UnableToCompleteException();
069            }
070            //Find Remote Service
071            JClassType remoteServiceType = typeOracle.findType(RemoteService.class.getName());
072            if (!serviceType.getFlattenedSupertypeHierarchy().contains(remoteServiceType)) {
073                logger.log(TreeLogger.ERROR, format("Type: %s not extends RemoteService", serviceName));
074                throw new UnableToCompleteException();
075            }
076        }

078        @Override
079        public String generate(TreeLogger logger, GeneratorContext context, String typeName) throws UnableToCompleteException {
080            this.context = context;
081            validate(logger, typeName);
082            this.className = format("%s_Proxy", this.sourceType.getSimpleSourceName());
083            generateClass(logger);
084            return this.packageName + "." + this.className;
085        }

087        private void generateClass(TreeLogger logger) {
088            PrintWriter printWriter = context.tryCreate(logger, this.packageName, this.className);
089            if (printWriter == null) {
090                return;
091            }
092            ClassSourceFileComposerFactory composer = new ClassSourceFileComposerFactory(this.packageName, this.className);
093            composer.addImport("com.google.gwt.core.client.GWT");
094            composer.addImport("com.google.gwt.core.client.Scheduler");
095            composer.addImport("com.google.gwt.event.shared.HandlerRegistration");
096            composer.addImport("com.google.gwt.user.client.rpc.AsyncCallback");
097            composer.addImport("com.gwt.ss.client.GwtAccessDeniedException");
098            composer.addImport("com.gwt.ss.client.GwtSecurityException");
099            composer.addImport("com.gwt.ss.client.loginable.HasLoginHandler");
100            composer.addImport("com.gwt.ss.client.loginable.LoginEvent");
101            composer.addImport("com.gwt.ss.client.loginable.LoginHandler");
102            composer.addImport("com.gwt.ss.client.loginable.LoginableService");
103            composer.addImport("com.gwt.ss.client.loginable.LoginCancelException");
104            composer.addImplementedInterface(this.typeName);
105            composer.addImplementedInterface(format("%s<%s>", LoginableService.class.getName(), this.typeName));
106            SourceWriter writer = composer.createSourceWriter(context, printWriter);
107            generateFields(writer);
108            generateMethods(writer);
109            writer.outdent();
110            writer.println("}");
111            context.commit(logger, printWriter);
112        }

146        private void generateMethods(SourceWriter writer) {
147            for (JMethod method : this.serviceType.getMethods()) {
148                generateMethod(writer, method);
149            }
150    
151            for (JClassType parentType : this.serviceType.getImplementedInterfaces()) {
152                for (JMethod method : parentType.getMethods()) {
153                    generateMethod(writer, method);
154                }
155            }
156        }

159            writer.println("@Override");

170            writer.println(", final AsyncCallback<%s> callback) {", method.getReturnType().getQualifiedSourceName());

177                } else {

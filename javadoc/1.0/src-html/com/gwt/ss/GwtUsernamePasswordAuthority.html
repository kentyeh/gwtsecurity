033    /**

043            ApplicationContextAware, ApplicationListener<InteractiveAuthenticationSuccessEvent> {

074        private PayloadInfo extract(JoinPoint jp) throws IOException, ServletException {
075            PayloadInfo result = payloadHolder.get();
076            HttpHolder httpHolder = HttpHolder.getInstance(jp);
077            HttpServletRequest request = httpHolder.getRequest();
078            String username = null;
079            String password = null;
080            if (result == null && request != null) {
081                String payload = RPCServletUtils.readContentAsGwtRpc(request);
082                if (logger.isDebugEnabled()) {
083                    logger.debug("login payload is [" + payload + "]");
084                }
085                StringTokenizer st = new StringTokenizer(payload, "|");
086                int idx = 0;
087                while (st.hasMoreTokens()) {
088                    switch (++idx) {
089                        case 9:
090                            username = st.nextToken();
091                            break;
092                        case 10:
093                            password = st.nextToken();
094                            break;
095                        default:
096                            st.nextToken();
097                    }
098                }
099                if (username != null && password != null) {
100                    result = new PayloadInfo(username, password, httpHolder);
101                    payloadHolder.set(result);
102                }
103            }
104            return result;
105        }
106    
107        @Around("execution(* org.springframework.security.web.authentication.AbstractAuthenticationProcessingFilter.doFilter(..))")
108        public Object doFilter(ProceedingJoinPoint pjp) throws Throwable {
109            HttpHolder httpHolder = HttpHolder.getInstance(pjp);
110            AbstractAuthenticationProcessingFilter filter = (AbstractAuthenticationProcessingFilter) pjp.getTarget();
111            if (httpHolder.isGwt() && requiresAuthentication(filter, httpHolder.getRequest())) {
112                PayloadInfo pi = null;
113                try {
114                    pi = extract(pjp);
115                    UsernamePasswordAuthenticationToken authRequest = new UsernamePasswordAuthenticationToken(pi.getUsername(), pi.getPassword());
116                    HttpSession session = httpHolder.getRequest().getSession(false);
117                    Class c = AbstractAuthenticationProcessingFilter.class;
118                    Field f = c.getDeclaredField("allowSessionCreation");
119                    f.setAccessible(true);
120                    boolean allowSessionCreation = (Boolean) f.get(filter);
121                    if (session != null || allowSessionCreation) {
122                        httpHolder.getRequest().getSession().setAttribute(GwtResponseUtil.SPRING_SECURITY_LAST_USERNAME_KEY,
123                                TextEscapeUtils.escapeEntities(pi.getUsername()));
124                    }
125                    authRequest.setDetails(filter.getAuthenticationDetailsSource().buildDetails(httpHolder.getRequest()));
126                    Authentication authResult = getAuthenticationManager().authenticate(authRequest);
127                    SecurityContextHolder.getContext().setAuthentication(authResult);
128                    filter.getRememberMeServices().loginSuccess(httpHolder.getRequest(), httpHolder.getResponse(), authResult);
129                    applicationContext.publishEvent(new InteractiveAuthenticationSuccessEvent(authResult, this.getClass()));
130                    return null;
131                } catch (Exception e) {
132                    if (logger.isErrorEnabled()) {
133                        logger.error("Gwt login fail:", e);
134                    }
135                    GwtResponseUtil.processGwtException(servletContext, httpHolder.getRequest(), httpHolder.getResponse(), e);
136                    return null;
137                } finally {
138                    payloadHolder.remove();
139                }
140            } else {
141                return pjp.proceed();
142            }
143        }
144    
145        protected boolean requiresAuthentication(AbstractAuthenticationProcessingFilter filter, HttpServletRequest request) {
146            String uri = request.getRequestURI();
147            if (logger.isDebugEnabled()) {
148                logger.debug("RequiresAuthentication check for \n"
149                        + "url = " + uri + "\ncontext path = " + request.getContextPath() + "\nprocessUrl = " + filter.getFilterProcessesUrl());
150            }
151            if (uri == null || uri.isEmpty()) {
152                return false;
153            } else {
154                int pathParamIndex = uri.indexOf(';');
155    
156                if (pathParamIndex > 0) {
157                    // strip everything after the first semi-colon
158                    uri = uri.substring(0, pathParamIndex);
159                }
160    
161                if ("".equals(request.getContextPath())) {
162                    return uri.endsWith(filter.getFilterProcessesUrl());
163                }
164                return uri.endsWith(request.getContextPath() + filter.getFilterProcessesUrl());
165            }
166        }
167    
168        @Override
169        public void onApplicationEvent(InteractiveAuthenticationSuccessEvent event) {
170            PayloadInfo pi = payloadHolder.get();
171            if (pi != null && pi.getHttpHolder().isGwt()) {
172                HttpServletRequest request = pi.getHttpHolder().getRequest();
173                Authentication authResult = event.getAuthentication();
174                if (logger.isDebugEnabled()) {
175                    logger.debug("Gwt authentication success. Updating SecurityContextHolder to contain: " + authResult);
176                }
177                GwtResponseUtil.writeResponse(servletContext, request, pi.getHttpHolder().getResponse(),
178                        String.format("//OK[[],%s,%s]", AbstractSerializationStream.DEFAULT_FLAGS,
179                        AbstractSerializationStream.SERIALIZATION_STREAM_VERSION));
180            }
181        }

193            }

204                return httpHolder;
205            }

208                this.httpHolder = httpHolder;
209            }
210        }
